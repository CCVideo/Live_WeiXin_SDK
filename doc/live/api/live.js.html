var StateMachine = require('./module/stateMachine');
var util = require('./module/util');
var Observer = require('./module/observer');
// var Net = require('./module/net');
var Login = require('./live/login');
var Pusher = require('./live/pusher');
var event = require('./live/event');
var History = require('./live/history');
var Player = require('./live/player');
var Manager = require('./live/manager');
var CONFIG = require('./module/config').LIVE;
var drawingBoard = require('./module/drawingBoard');

var stateMachine = new StateMachine();
var observer = new Observer(CONFIG);
drawingBoard.init(observer);
var history = new History(stateMachine, observer);
var player = new Player(stateMachine, observer);
var manager = new Manager(stateMachine, observer);
// var net = new Net(stateMachine);

var reconnectParam = {};

/**
 * @description 初始化
 * @param {Object} param 初始化参数对象
 * @property {String} param.userId 用户 id
 * @property {String} param.roomId 直播间id
 * @property {String} param.userName 用户名称
 * @property {String} param.password 需要密码验证方式时，可选
 * @property {Object} param.wx 微信全局对象
 * @property {Boolean} param.forcibly 强制登录，可选
 * @property {Function} param.success 初始化成功回掉函数，可选
 * @property {Function} param.fail 初始化失败回掉函数，可选
 * @example cc.live.init({
    userId: 'userId',
    roomId: 'roomId',
    userName: 'userName',
    password: 'password',
    wx: wx,
    forcibly: false,
    success: function (data) {
      console.log('init success',data);
    },
    fail: function (data) {
      console.log('init fail',data);
    }
  });
 */

var init = function (param, reconnect) {

    reconnectParam = param;

    var login = new Login(stateMachine);

    login.login(param, function () {

        //网路状态变化处理
        // net.onNetworkStatusChange(init, login.getParam(), manager);

        var pusher = new Pusher(stateMachine, manager, observer);

        pusher.connectIo(login.getParam(), login.getData());

        pusher.getIo().on('authorized', function () {

            //配置播放地址
            player.init(login.getParam(), login.getData());

            if (stateMachine.isFirstListenerEvents()) {
                stateMachine.setFirstListenerEvents(false);
                //监听事件
                event.ListenerEvent(login.getData(), pusher.getIo(), stateMachine, observer);
            }

            //方法
            manager.init(login, pusher, player, history);

            //获得历史信息
            history.init(login.getParam(), reconnect, login.getData());

        });

    }, reconnect);

};

var reconnect = function (callback) {

    if (JSON.stringify(reconnectParam) === '{}') {
        util.log('reconnect error no init', reconnectParam);
        return false;
    }

    if (callback && typeof callback !== 'function') {
        util.log('reconnect error param typeof no callback', callback);
        return false;
    }

    var opts = {
        isReconnect: true,
        callback: callback
    };

    init(reconnectParam, opts);
};

var onNetworkStatusChange = function (callback) {
    stateMachine.setOnNetworkStatusChange(true);
    if (stateMachine.isFirstOnNetworkStatusChange()) {
        stateMachine.setFirstOnNetworkStatusChange(false);
        wx.onNetworkStatusChange(function (state) {
            if (stateMachine.isOnNetworkStatusChange()) {
                if (state.isConnected) {
                    reconnect(callback);
                } else {
                    manager.quit();
                }
            }
        });
    }
};
